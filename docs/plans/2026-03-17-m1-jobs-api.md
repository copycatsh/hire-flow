# M1 Jobs API Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Full CRUD for jobs and profiles with Outbox pattern publishing events to NATS JetStream.

**Architecture:** Flat Go package structure inside `services/jobs-api/`. Handlers own transaction boundaries (BEGIN → write entity + outbox → COMMIT). Background outbox publisher goroutine polls PostgreSQL every 5s, publishes to NATS JOBS stream. Graceful shutdown via context cancellation + wg.Go(). pgx for database, goose for migrations (embedded SQL), testcontainers for integration tests.

**Tech Stack:** Go 1.25, Echo v4, pgx v5 (pgxpool), goose v3, nats.go (jetstream), testcontainers-go, testify

**Decisions from eng review:**
- DB driver: pgx v5 (pgxpool for connection pool)
- Migrations: goose v3 with embedded SQL (runs from main.go on startup)
- Skills: separate `skills` table + `profile_skills` join table
- Tests: testcontainers-go for integration tests
- Outbox: local to jobs-api (extract to pkg/ in M3)
- Outbox lifecycle: context + wg.Go() for graceful shutdown
- NATS: JOBS stream, subjects `jobs.job.created`, `jobs.job.updated`, `jobs.profile.updated`
- TX ownership: handler owns transaction boundary
- Store interface: DBTX interface (pool + tx compatible)
- Error handling: custom Echo error handler in middleware.go
- Outbox poll interval: configurable via env var, default 5s

```

        │
        ▼
   job_handler.go
        │
   pool.Begin(ctx)
        │
   ┌────┴────┐
   │  INSERT  │  INSERT
   │  jobs    │  outbox (job.created)
   └────┬────┘
        │

        │
        ▼
   201 Created
                    ┌─────────────────┐
```

```go
package main

import (
	"context"
	"fmt"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
)

type PostgresJobStore struct{}

func (s *PostgresJobStore) Create(ctx context.Context, db DBTX, req CreateJobRequest) (Job, error) {
	var j Job
	err := db.QueryRow(ctx,
		`INSERT INTO jobs (title, description, budget_min, budget_max, client_id)
		 VALUES ($1, $2, $3, $4, $5)
		 RETURNING id, title, description, budget_min, budget_max, status, client_id, created_at, updated_at`,
		req.Title, req.Description, req.BudgetMin, req.BudgetMax, req.ClientID,
	).Scan(&j.ID, &j.Title, &j.Description, &j.BudgetMin, &j.BudgetMax, &j.Status, &j.ClientID, &j.CreatedAt, &j.UpdatedAt)
	if err != nil {
		return Job{}, fmt.Errorf("insert job: %w", err)
	}
	return j, nil
}

func (s *PostgresJobStore) GetByID(ctx context.Context, db DBTX, id uuid.UUID) (Job, error) {
	var j Job
	err := db.QueryRow(ctx,
		`SELECT id, title, description, budget_min, budget_max, status, client_id, created_at, updated_at
		 FROM jobs WHERE id = $1`,
		id,
	).Scan(&j.ID, &j.Title, &j.Description, &j.BudgetMin, &j.BudgetMax, &j.Status, &j.ClientID, &j.CreatedAt, &j.UpdatedAt)
	if err != nil {
		return Job{}, fmt.Errorf("get job: %w", err)
	}
	return j, nil
}

func (s *PostgresJobStore) List(ctx context.Context, db DBTX, params ListJobsParams) ([]Job, error) {
	query := `SELECT id, title, description, budget_min, budget_max, status, client_id, created_at, updated_at
	          FROM jobs`
	args := []any{}
	argIdx := 1

	if params.Status != "" {
		query += fmt.Sprintf(" WHERE status = $%d", argIdx)
		args = append(args, params.Status)
		argIdx++
	}

	query += " ORDER BY created_at DESC"

	if params.Limit > 0 {
		query += fmt.Sprintf(" LIMIT $%d", argIdx)
		args = append(args, params.Limit)
		argIdx++
	}
	if params.Offset > 0 {
		query += fmt.Sprintf(" OFFSET $%d", argIdx)
		args = append(args, params.Offset)
	}

	rows, err := db.Query(ctx, query, args...)
	if err != nil {
		return nil, fmt.Errorf("list jobs: %w", err)
	}
	defer rows.Close()

	return pgx.CollectRows(rows, pgx.RowToStructByPos[Job])
}

func (s *PostgresJobStore) Update(ctx context.Context, db DBTX, id uuid.UUID, req UpdateJobRequest) (Job, error) {
	var j Job
	err := db.QueryRow(ctx,
		`UPDATE jobs SET
			title = COALESCE($2, title),
			description = COALESCE($3, description),
			budget_min = COALESCE($4, budget_min),
			budget_max = COALESCE($5, budget_max),
			status = COALESCE($6, status),
			updated_at = now()
		 WHERE id = $1
		 RETURNING id, title, description, budget_min, budget_max, status, client_id, created_at, updated_at`,
		id, req.Title, req.Description, req.BudgetMin, req.BudgetMax, req.Status,
	).Scan(&j.ID, &j.Title, &j.Description, &j.BudgetMin, &j.BudgetMax, &j.Status, &j.ClientID, &j.CreatedAt, &j.UpdatedAt)
	if err != nil {
		return Job{}, fmt.Errorf("update job: %w", err)
	}
	return j, nil
}
```

```go
		`INSERT INTO outbox (aggregate_type, aggregate_id, event_type, payload)
		 VALUES ($1, $2, $3, $4)`,

		return fmt.Errorf("insert outbox: %w", err)
	}
	return nil
}

func (s *PostgresOutboxStore) FetchUnpublished(ctx context.Context, db DBTX, limit int) ([]OutboxEntry, error) {
	rows, err := db.Query(ctx,
		`SELECT id, aggregate_type, aggregate_id, event_type, payload, created_at, published_at
		 FROM outbox
		 WHERE published_at IS NULL
		 ORDER BY created_at ASC
		 LIMIT $1
		 FOR UPDATE SKIP LOCKED`,
		limit,
	)
	if err != nil {
		return nil, fmt.Errorf("fetch outbox: %w", err)
	}
	defer rows.Close()

	return pgx.CollectRows(rows, pgx.RowToStructByPos[OutboxEntry])
}

func (s *PostgresOutboxStore) MarkPublished(ctx context.Context, db DBTX, ids []uuid.UUID) error {
	_, err := db.Exec(ctx,
		`UPDATE outbox SET published_at = now() WHERE id = ANY($1)`,
		ids,
	)
	if err != nil {
		return fmt.Errorf("mark published: %w", err)
	}
	return nil
}
```

```go
	}
	return p, nil

	var p Profile
	err := db.QueryRow(ctx,
		`SELECT id, full_name, bio, hourly_rate, available, created_at, updated_at
		 FROM profiles WHERE id = $1`,
		id,
	).Scan(&p.ID, &p.FullName, &p.Bio, &p.HourlyRate, &p.Available, &p.CreatedAt, &p.UpdatedAt)
	if err != nil {
		return Profile{}, fmt.Errorf("get profile: %w", err)
	}

	// Load skills via join table
	rows, err := db.Query(ctx,
		`SELECT s.id, s.name, s.category, s.created_at
		 FROM skills s
		 JOIN profile_skills ps ON ps.skill_id = s.id
		 WHERE ps.profile_id = $1
		 ORDER BY s.name`,
		id,
	)
	if err != nil {
		return Profile{}, fmt.Errorf("get profile skills: %w", err)
	}
	defer rows.Close()

	skills, err := pgx.CollectRows(rows, pgx.RowToStructByPos[Skill])
	if err != nil {
		return Profile{}, fmt.Errorf("scan profile skills: %w", err)
	}
	p.Skills = skills

	return p, nil
}

func (s *PostgresProfileStore) UpdateSkills(ctx context.Context, db DBTX, profileID uuid.UUID, skillIDs []uuid.UUID) error {
	// Delete existing skills
	_, err := db.Exec(ctx, `DELETE FROM profile_skills WHERE profile_id = $1`, profileID)
	if err != nil {
		return fmt.Errorf("delete profile skills: %w", err)
	}

	// Insert new skills
	for _, skillID := range skillIDs {
		_, err := db.Exec(ctx,
			`INSERT INTO profile_skills (profile_id, skill_id) VALUES ($1, $2)`,
			profileID, skillID,
		)
		if err != nil {
			return fmt.Errorf("insert profile skill: %w", err)
		}
	}

	// Update profile timestamp
	_, err = db.Exec(ctx, `UPDATE profiles SET updated_at = now() WHERE id = $1`, profileID)
	if err != nil {
		return fmt.Errorf("update profile timestamp: %w", err)
	}

	return nil
}
```

```go
// file: services/jobs-api/middleware.go
package main

import (
	"errors"
	"log/slog"
	"net/http"

	"github.com/jackc/pgx/v5"
	"github.com/labstack/echo/v4"
)

// AppError is a sentinel error type for known application errors.
type AppError struct {
	Code    int    `json:"-"`
	Message string `json:"message"`
}

func (e *AppError) Error() string { return e.Message }

func NewAppError(code int, msg string) *AppError {
	return &AppError{Code: code, Message: msg}
}

// customErrorHandler maps errors to HTTP responses.
// pgx.ErrNoRows → 404, AppError → its code, default → 500.
func customErrorHandler(err error, c echo.Context) {
	if c.Response().Committed {
		return
	}

	var appErr *AppError
	switch {
	case errors.As(err, &appErr):
		_ = c.JSON(appErr.Code, map[string]string{"error": appErr.Message})
	case errors.Is(err, pgx.ErrNoRows):
		_ = c.JSON(http.StatusNotFound, map[string]string{"error": "not found"})
	default:
		var echoErr *echo.HTTPError
		if errors.As(err, &echoErr) {
			_ = c.JSON(echoErr.Code, map[string]string{"error": http.StatusText(echoErr.Code)})
			return
		}
		slog.Error("unhandled error", "error", err, "path", c.Path())
		_ = c.JSON(http.StatusInternalServerError, map[string]string{"error": "internal server error"})
	}
}

func requestLogger() echo.MiddlewareFunc {
	return func(next echo.HandlerFunc) echo.HandlerFunc {
		return func(c echo.Context) error {
			err := next(c)
			req := c.Request()
			res := c.Response()
			slog.Info("request",
				"method", req.Method,
				"path", req.URL.Path,
				"status", res.Status,
			)
			return err
		}
	}
}
```

```go
type JobHandler struct {

func (h *JobHandler) Create(c echo.Context) error {
	var req CreateJobRequest
	if err := c.Bind(&req); err != nil {

	}
	if req.Title == "" {
```

```go
// --- append to outbox.go after OutboxStore interface ---

// OutboxPublisher polls the outbox table and publishes events to NATS.
//
// Lifecycle:
//   ctx cancel → stop polling → drain in-flight batch → return
//
//                    ┌──────────────────────┐
//                    │    Poll Loop         │
//                    │  ┌────────────────┐  │
//   ctx.Done() ──────┤  │ FetchUnpublished│  │
//                    │  │ (batch 100)    │  │
//                    │  └───────┬────────┘  │
//                    │          │            │
//                    │  ┌───────▼────────┐  │
//                    │  │ Publish to NATS│  │
//                    │  │ (per entry)    │  │
//                    │  └───────┬────────┘  │
//                    │          │            │
//                    │  ┌───────▼────────┐  │
//                    │  │ MarkPublished  │  │
//                    │  │ (batch)        │  │
//                    │  └────────────────┘  │
//                    │       sleep(interval)│
//                    └──────────────────────┘
type OutboxPublisher struct {
	store    OutboxStore
	pool     DBTX
	nats     *NATSClient
	interval time.Duration
	batch    int
}

func NewOutboxPublisher(store OutboxStore, pool DBTX, nats *NATSClient, interval time.Duration) *OutboxPublisher {
	return &OutboxPublisher{
		store:    store,
		pool:     pool,
		nats:     nats,
		interval: interval,
		batch:    100,
	}
}

func (p *OutboxPublisher) Run(ctx context.Context) {
	slog.Info("outbox publisher started", "interval", p.interval)
	ticker := time.NewTicker(p.interval)
	defer ticker.Stop()

	for {
		select {
		case <-ctx.Done():
			slog.Info("outbox publisher stopping")
			return
		case <-ticker.C:
			if err := p.publishBatch(ctx); err != nil {
				slog.Error("outbox publish batch failed", "error", err)
			}
		}
	}
}

func (p *OutboxPublisher) publishBatch(ctx context.Context) error {
	entries, err := p.store.FetchUnpublished(ctx, p.pool, p.batch)
	if err != nil {
		return fmt.Errorf("fetch unpublished: %w", err)
	}
	if len(entries) == 0 {
		return nil
	}

	published := make([]uuid.UUID, 0, len(entries))
	for _, e := range entries {
		if err := p.nats.Publish(ctx, e.EventType, e.Payload); err != nil {
			slog.Error("failed to publish event",
				"event_type", e.EventType,
				"aggregate_id", e.AggregateID,
				"error", err,
			)
			continue
		}
		published = append(published, e.ID)
	}

	if len(published) > 0 {
		if err := p.store.MarkPublished(ctx, p.pool, published); err != nil {
			return fmt.Errorf("mark published: %w", err)
		}
		slog.Info("outbox published", "count", len(published))
	}

	return nil
}
```

```go
func main() {
	ctx, stop := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer stop()

	// --- Config ---
	dbURL := cmp.Or(os.Getenv("DATABASE_URL"), "postgres://hire_flow:hire_flow_dev@localhost:5432/jobs_db?sslmode=disable")
	natsURL := cmp.Or(os.Getenv("NATS_URL"), "nats://localhost:4222")
	port := cmp.Or(os.Getenv("PORT"), ":8001")
	pollInterval, err := time.ParseDuration(cmp.Or(os.Getenv("OUTBOX_POLL_INTERVAL"), "5s"))
	if err != nil {
		slog.Error("invalid OUTBOX_POLL_INTERVAL", "error", err)
		os.Exit(1)
	}

	// --- Run migrations (goose via database/sql) ---
	sqlDB, err := sql.Open("pgx", dbURL)
	if err != nil {
		slog.Error("open sql for migrations", "error", err)
		os.Exit(1)
	}
	goose.SetBaseFS(embedMigrations)
	if err := goose.SetDialect("postgres"); err != nil {
		slog.Error("set goose dialect", "error", err)
		os.Exit(1)
	}
	if err := goose.Up(sqlDB, "migrations"); err != nil {
		slog.Error("run migrations", "error", err)
		os.Exit(1)
	}
	sqlDB.Close()
	slog.Info("migrations applied")

	// --- Database pool (pgx) ---
	pool, err := pgxpool.New(ctx, dbURL)
	if err != nil {
		slog.Error("create db pool", "error", err)
		os.Exit(1)
	}
	defer pool.Close()

	// --- NATS ---
	nc, err := NewNATSClient(natsURL)
	if err != nil {
		slog.Error("connect nats", "error", err)
		os.Exit(1)
	}
	defer nc.Close()

	if err := nc.EnsureStream(ctx); err != nil {
		slog.Error("ensure nats stream", "error", err)
		os.Exit(1)
	}

	// --- Stores ---
	jobStore := &PostgresJobStore{}
	profileStore := &PostgresProfileStore{}
	outboxStore := &PostgresOutboxStore{}

	// --- Echo ---
	e := echo.New()
	e.HideBanner = true
	e.HTTPErrorHandler = customErrorHandler
	e.Use(requestLogger())

	e.GET("/health", func(c echo.Context) error {
		return c.JSON(http.StatusOK, map[string]string{"status": "ok"})
	})

	jobHandler := &JobHandler{pool: pool, jobs: jobStore, outbox: outboxStore}
	jobHandler.RegisterRoutes(e.Group("/api/v1/jobs"))

	profileHandler := &ProfileHandler{pool: pool, profiles: profileStore, outbox: outboxStore}
	profileHandler.RegisterRoutes(e.Group("/api/v1/profiles"))

	// --- Outbox publisher (background goroutine) ---
	publisher := NewOutboxPublisher(outboxStore, pool, nc, pollInterval)
	var wg sync.WaitGroup
	wg.Go(func() {
		publisher.Run(ctx)
	})

	// --- Start server ---
	slog.Info("starting jobs-api", "port", port)
	go func() {
		if err := e.Start(port); err != nil && err != http.ErrServerClosed {
			slog.Error("server failed", "error", err)
			os.Exit(1)
		}
	}()

	// --- Graceful shutdown ---
	<-ctx.Done()
	slog.Info("shutting down")

	shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
	defer cancel()
	if err := e.Shutdown(shutdownCtx); err != nil {
		slog.Error("echo shutdown", "error", err)
	}
	wg.Wait()
	slog.Info("shutdown complete")
}
```

```go
func TestOutboxPublisher_Integration(t *testing.T) {
	if testing.Short() {
		t.Skip("skipping integration test")
	}

	ctx := t.Context()
	pool, pgCleanup := setupPostgres(t, ctx)
	defer pgCleanup()
	nc, natsCleanup := setupNATS(t, ctx)
	defer natsCleanup()

	outboxStore := &PostgresOutboxStore{}
	jobStore := &PostgresJobStore{}

	// Create a job (writes outbox entry)
	tx, err := pool.Begin(ctx)
	require.NoError(t, err)

	job, err := jobStore.Create(ctx, tx, CreateJobRequest{
		Title:    "Test Job",
		ClientID: uuid.New(),
	})
	require.NoError(t, err)

	payload, _ := json.Marshal(job)
	require.NoError(t, outboxStore.Insert(ctx, tx, OutboxEntry{
		AggregateType: "job",
		AggregateID:   job.ID,
		EventType:     "jobs.job.created",
		Payload:       payload,
	}))
	require.NoError(t, tx.Commit(ctx))

	// Subscribe to NATS to verify event arrives
	rawNC, err := nats.Connect(nc.conn.ConnectedUrl())
	require.NoError(t, err)
	js, err := jetstream.New(rawNC)
	require.NoError(t, err)

	cons, err := js.CreateOrUpdateConsumer(ctx, "JOBS", jetstream.ConsumerConfig{
		FilterSubject: "jobs.job.created",
		AckPolicy:     jetstream.AckExplicitPolicy,
	})
	require.NoError(t, err)

	// Run one publish cycle
	publisher := NewOutboxPublisher(outboxStore, pool, nc, time.Second)
	require.NoError(t, publisher.publishBatch(ctx))

	// Verify event received
	msg, err := cons.Next(jetstream.FetchMaxWait(5 * time.Second))
	require.NoError(t, err)
	assert.Contains(t, string(msg.Data()), "Test Job")

	// Verify outbox marked as published
	var publishedAt *time.Time
	err = pool.QueryRow(ctx, "SELECT published_at FROM outbox WHERE aggregate_id = $1", job.ID).Scan(&publishedAt)
	require.NoError(t, err)
	assert.NotNil(t, publishedAt)
}
```
